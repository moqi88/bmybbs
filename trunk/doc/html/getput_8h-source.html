00055 /*------------ macros for storing/extracting msb first words -------------*/

00057 #define GET_32BIT(cp) (((unsigned long)(unsigned char)(cp)[0] << 24) | \
00058                        ((unsigned long)(unsigned char)(cp)[1] << 16) | \
00059                        ((unsigned long)(unsigned char)(cp)[2] << 8) | \
00060                        ((unsigned long)(unsigned char)(cp)[3]))
00061 
00062 #define GET_16BIT(cp) (((unsigned long)(unsigned char)(cp)[0] << 8) | \
00063                        ((unsigned long)(unsigned char)(cp)[1]))
00064 
00065 #define PUT_32BIT(cp, value) do { \
00066   (cp)[0] = (value) >> 24; \
00067   (cp)[1] = (value) >> 16; \
00068   (cp)[2] = (value) >> 8; \
00069   (cp)[3] = (value); } while (0)

00071 #define PUT_16BIT(cp, value) do { \
00072   (cp)[0] = (value) >> 8; \
00073   (cp)[1] = (value); } while (0)
