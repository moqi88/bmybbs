00055 #ifndef RANDOM_H
00056 #define RANDOM_H
00057 
00058 #include "md5.h"
00059 
00060 #define RANDOM_STATE_BITS       8192
00061 #define RANDOM_STATE_BYTES      (RANDOM_STATE_BITS / 8)
00062 
00063 /* Structure for the random state. */
00064 typedef struct {
00065     unsigned char state[RANDOM_STATE_BYTES];    /* Pool of random data. */
00066     unsigned char stir_key[64]; /* Extra data for next stirring. */
00067     unsigned int next_available_byte;   /* Index of next available byte. */
00068     unsigned int add_position;  /* Index to add noise. */
00069     time_t last_dev_random_usage;       /* Time of last /dev/random usage. */
00070 } RandomState;
00071 
00072 /* Initializes the random number generator, loads any random information
00073    from the given file, and acquires as much environmental noise as it
00074    can to initialize the random number generator.  More noise can be
00075    acquired later by calling random_add_noise + random_stir, or by
00076    calling random_get_environmental_noise again later when the environmental
00077    situation has changed.  All I/O will be done with the given uid. */
00078 void random_initialize(RandomState * state, uid_t uid, const char *filename);
00079 
00080 /* Acquires as much environmental noise as it can.  This is probably quite
00081    sufficient on a unix machine, but might be grossly inadequate on a
00082    single-user PC or a Macintosh.  This call random_stir automatically. 
00083    This call may take many seconds to complete on a busy system. 
00084    This will perform any commands with the given uid using userfile. */
00085 void random_acquire_environmental_noise(RandomState * state, uid_t uid);
00086 
00087 /* Acquires easily available noise from the environment. */
00088 void random_acquire_light_environmental_noise(RandomState * state);
00089 
00090 /* Executes the given command, and processes its output as noise.
00091    random_stir should be called after this.  The command will be called
00092    with the given uid via userfile. */
00093 void random_get_noise_from_command(RandomState * state, uid_t uid, const char *cmd);
00094 
00095 /* Adds the contents of the buffer as noise.  random_stir should be called
00096    after this. */
00097 void random_add_noise(RandomState * state, const void *buf, unsigned int bytes);
00098 
00099 /* Stirs the random pool to consume any newly acquired noise or to get more
00100    random numbers.  This should be called after adding noise to properly
00101    mix the noise into the random pool. */
00102 void random_stir(RandomState * state);
00103 
00104 /* Returns a random byte.  Stirs the random pool if necessary.  Acquires
00105    new environmental noise approximately every five minutes. */
00106 unsigned int random_get_byte(RandomState * state);
00107 
00108 /* Saves some random bits in the file so that it can be used as a source
00109    of randomness for later runs.  I/O will be done with the given uid using
00110    userfile. */
00111 void random_save(RandomState * state, uid_t uid, const char *filename);
00112 
00113 /* Zeroes and frees any data structures associated with the random number
00114    generator. */
00115 void random_clear(RandomState * state);
00116 
00117 #endif                          /* RANDOM_H */
